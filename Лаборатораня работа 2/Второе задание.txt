using System;

public class HelloWorld
{
    public static void Main(string[] args)
    {
        int n = Convert.ToInt32(Console.ReadLine());
        int k = 0;
        int maxk = 0;
        while (n>0){
            n = n - 1;
            int num = Convert.ToInt32(Console.ReadLine());
            if (num % 2 == 0){
                k += num;
            }
            else{ 
                if (maxk<k){
                    maxk = k;
                    k = 0;
                }
            }
        }
        
            Console.WriteLine(maxk);
        }
    }
